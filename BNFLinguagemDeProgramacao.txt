<especificadora-de-tipo> ::= void | char | int | double | short | long | String

<operador-atribuicao> ::= = | *= | /= | %= | += | -= | <<= | >>= | &= | ^= | |=

<operador-unitario> ::= & | * | + | - | ~ | !

//declaração
<declaracao> ::=

<declaracao-selecao> ::= 
		if (<expressao>) <declaracao>
		| if (<expressao>) <declaracao> else <declaracao>
		| switch (<expressao>) <declaracao>

<declaracao-iteracao> ::= 
		while ( <expressao> ) <declaracao>
		| do <declaracao> while ( <expressao> ) ;
		| for ({<expressao>}?; {<expressao>}?; {<expressao>}?) <declaracao>
