Resumo:
" ": Terminal
[ ]: Opcional
( ): Agrupamento
{ }: Repetição

- Tipagem e Variáveis

tipoVariavel = "void" | "char" | "short" | "int" | "long" | "float" | "double"
declaracaoVariavel = tipoVariavel identificador ";" // caso 1, apenas declarou
declaracaoVariavel = tipoVariavel identificador [ "=" expressao ] ";" // caso 2, declarou e atribuiu
 
- Atribuição

declaracaoAtribuicao = identificador "=" expressao ";" 

- Estruturas de Controle

	- Condicional

	declaracaoSelecao = "if" "(" expressao ")" comando ["else" comando]

	- Repetição

	declaracaoIteracao = "while" "(" expressao ")" comando | "for" "(" [atribuicao] ";" [expressao] ";" [atribuicao] ")" comando | 
						 "do" comando "while" "(" expressao ")" ";"

- Expressões

operadorRelacional = "<" | ">" | "<=" | ">=" | "==" | "!="
